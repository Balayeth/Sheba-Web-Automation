package org.example;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

import java.time.Duration;

public class Main {



    public static void main(String[] args) {
        WebDriver driver;
        driver = new ChromeDriver();
        driver.manage().window().maximize();
        driver.get("https://distibutor-web-design.vercel.app/");
        driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(20));

        By phoneNo = By.xpath("//input[@id='number']");
        driver.findElement(phoneNo).sendKeys("01581578235");

        By submit = By.xpath("//button[@type='submit']");
        driver.findElement(submit).click();

        By pin = By.xpath("//input[@id='PIN']");
        driver.findElement(pin).sendKeys("12345");

        driver.findElement(submit).click();

        By productList = By.xpath("(//p[normalize-space()='প্রোডাক্ট লিস্ট'])[1]");
        driver.findElement(productList).isDisplayed();
        driver.findElement(productList).click();

        By addProduct = By.xpath("//button[contains(text(),'পণ্য যোগ করুন')]");
        driver.findElement(addProduct).click();


    }

}
